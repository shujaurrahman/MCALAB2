        mul=new JButton("Mul");

        panel.add(numLabel1);

        panel.add(num2);
        panel.add(add);
        panel.add(sub);
        panel.add(mul);
        panel.add(div);
        panel.add(result);
        fr.add(panel);
        fr.setVisible(true);
    }
    public void actionPerformed(ActionEvent e){
        int n1=Integer.parseInt(num1.getText());
        int n2=Integer.parseInt(num2.getText());
        if(e.getSource()==add){
            int r=n1+n2;
            result.setText("Result of Addition = "+r);
        }else if(e.getSource()==sub){
            int r=n1-n2;
            result.setText("Result of Subtraction = "+r);
        }else if(e.getSource()==mul){
            int r=n1*n2;
            result.setText("Result of Multiplication = "+r);
        }else if(e.getSource()==div){
            int r;
            if(n2==0){
                result.setText("Division not possible. n2 cannot be zero");
            }else{
                r=n1/n2;
                result.setText("Result of Division = "+r);
            }
        }else{
            result.setText("Some error occured!");
        }
    }
    public static void main(String[] args) {
        SwingUtilities.invokeLater(exam::new);
    }
}

        Scanner sc=new Scanner(System.in);
        int[] arr=new int[5];
        for (int i=0;i<5;i++){
            arr[i]=sc.nextInt();
        }
        //Sort array in decending order
        for (int i=0;i<5;i++){
            for (int j=i+1;j<5;j++){
                if(arr[j]>arr[i]){
                    int swap=arr[i];
                    arr[i]=arr[j];
                    arr[j]=swap;
                }
            }

        }
        for(int e:arr){
            System.out.print(e+" ");
        }
        System.out.println("Second Largest Element in an array is : "+arr[1]);
    }

class ISBalance extends Exception{
    public ISBalance(String message){
        super(message);
    }
}
class Bank{
    private double account;
    private double remAmount;

    public Bank(double account,double remAmount){
        this.account=account;
        this.remAmount=remAmount;
    }

    public double Balance(){
        return remAmount;
    }
    public void deposit(double amount){
        remAmount+=amount;
        System.out.println("Deposit: "+amount);

    }

    public void withdraw(double amount) throws ISBalance{
        if(amount>remAmount){
            throw new ISBalance("Insufficient Balance in account");
        }else{
            remAmount-=amount;
        }
    }
}

import java.util.Scanner;

public class exam {
    public static boolean isPrime(int num){
        if(num<2) return false;
        for(int i=2;i<=Math.sqrt(num);i++){
            if(num%i==0){
                return false;
            }
        }
        return true;
    }
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        while(true){
        int n=sc.nextInt();
        if(isPrime(n)){
            System.out.println("Entered Number is a prime Number. ");
            break;
        }else{
            System.out.println("Entered Number is not a prime Number");
        }
    }}
}

public class exam {

        if(n<=1){
           return n;
        }else{
            return fib(n-1)+fib(n-2);
        }
    }

        int n=sc.nextInt();

        }
    }
}

import java.util.Scanner;
import java.util.StringTokenizer;

public class exam {
    public static void main(String[] arg){
        Scanner sc=new Scanner(System.in);
        String text=sc.nextLine();
        StringTokenizer st=new StringTokenizer(text);

        int wordCount=0;
        int sentenceCount=0;
        int Ecount=0;
        int Zcount=0;

        while(st.hasMoreElements()){
            String word=st.nextToken();
            wordCount++;

            char lastChar=word.charAt(word.length()-1);
            if(lastChar=='.'|| lastChar=='!' || lastChar=='?'){
                sentenceCount++;
            }
            for (int i=0;i<word.length();i++){
                char ch=word.charAt(i);
                if(ch=='e' || ch=='E'){
                    Ecount++;
                }
                if(ch=='z' || ch=='Z'){
                    Zcount++;
                }
            }
        }

        System.out.println(wordCount+" "+sentenceCount+" "+Ecount+" "+Zcount);
    }
}
